#include <iostream>
#include <map>
#include <string>
#include <cstdlib> // Для работы с функцией system()
using namespace std;

int main()
{
	map<string, double> units = { { "km", 1E3 },{ "m", 1 },{ "dm", 1E-1 },{ "mm", 1E-3 },{ "mkm", 1E-6 },{ "nm", 1E-9 } };

	double x;
	string unit;

	cout << "Dlina: ";
	cin >> x >> unit;

	for (auto i : units)
	{
		if (i.first != unit)
			cout << (x * units[unit] / i.second) << " " << i.first << endl;
	}
	system("pause"); // Команда задержки экрана
	return 0;
}
